<!DOCTYPE html>
<html lang="ja">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>K-STEP 体調管理シート</title>
    <script src="https://unpkg.com/react@18/umd/react.development.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <script src="https://cdn.tailwindcss.com"></script>
    <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/lucide-static@0.321.0/font/lucide.min.css">
    <style>
        th { writing-mode: vertical-rl; height: 120px; font-weight: normal; }
        .no-vertical { writing-mode: horizontal-tb; height: auto; }
    </style>
</head>
<body class="bg-slate-50">
    <div id="root"></div>

    <script type="text/babel">
        const { useState } = React;

        const conditionItems = ['気分', '睡眠', '食事', 'ストレス'];
        const goodSigns = ['まあいいか', '心に余裕がある', '適度な身だしなみ', '入浴が心地よい', '同じ時間に起きる', '適度な寝だめ'];
        const warningSigns = ['満腹感がない', '過眠', '何も考えられない', '言葉数が少ない', 'やけ食い', '字が乱雑になる', '収納癖が出る'];
        const badSigns = ['眠気を全く感じない', '物音に過敏', '家族を敵視', '散財・遠方へ行く', 'やることが多い', '脱着衣が激しい', '全能感'];
        const recoveryItems = ['コーヒー', 'スキンケア', 'マインドフルネス'];

        function App() {
            const [records, setRecords] = useState([
                { id: 1, date: new Date().toISOString().split('T')[0], time: '朝', conditions: {}, goodSigns: {}, warningSigns: {}, badSigns: {}, recovery: {}, medication: false, note: '' }
            ]);

            const addRecord = () => {
                const last = records[records.length - 1];
                let nextTime = last?.time === '朝' ? '昼' : last?.time === '昼' ? '晩' : '朝';
                setRecords([...records, { id: Date.now(), date: last?.date || '', time: nextTime, conditions: {}, goodSigns: {}, warningSigns: {}, badSigns: {}, recovery: {}, medication: false, note: '' }]);
            };

            const toggle = (id, cat, item) => {
                setRecords(records.map(r => r.id === id ? { ...r, [cat]: { ...r[cat], [item]: !r[cat][item] } } : r));
            };

            const update = (id, field, val, sub) => {
                setRecords(records.map(r => r.id === id ? (sub ? { ...r, [field]: { ...r[field], [sub]: val } } : { ...r, [field]: val }) : r));
            };

            return (
                <div className="p-4 max-w-[1400px] mx-auto">
                    <div className="flex justify-between items-center mb-6 bg-white p-4 rounded-xl shadow-sm">
                        <h1 className="text-xl font-bold">K-STEP 体調管理シート</h1>
                        <button onClick={addRecord} className="bg-indigo-600 text-white px-4 py-2 rounded-lg text-sm hover:bg-indigo-700">＋ 行を追加</button>
                    </div>
                    
                    <div className="overflow-x-auto bg-white rounded-xl shadow-md border border-slate-200">
                        <table className="w-max border-collapse">
                            <thead>
                                <tr className="bg-slate-800 text-white text-[10px]">
                                    <th className="no-vertical p-2 border border-slate-700 min-w-[120px]">日付</th>
                                    <th className="no-vertical p-2 border border-slate-700 w-16">時間</th>
                                    {conditionItems.map(h => <th key={h} className="border border-slate-700 p-1">{h}</th>)}
                                    {goodSigns.map(h => <th key={h} className="border border-slate-700 p-1 bg-blue-900/30">{h}</th>)}
                                    {warningSigns.map(h => <th key={h} className="border border-slate-700 p-1 bg-amber-900/30">{h}</th>)}
                                    {badSigns.map(h => <th key={h} className="border border-slate-700 p-1 bg-red-900/30">{h}</th>)}
                                    {recoveryItems.map(h => <th key={h} className="border border-slate-700 p-1 bg-green-900/30">{h}</th>)}
                                    <th className="no-vertical p-2 border border-slate-700 w-12">頓服</th>
                                    <th className="no-vertical p-2 border border-slate-700 min-w-[150px]">備考</th>
                                </tr>
                            </thead>
                            <tbody>
                                {records.map(r => (
                                    <tr key={r.id} className="hover:bg-slate-50 border-b border-slate-100">
                                        <td className="p-1 border border-slate-200"><input type="date" value={r.date} onChange={e => update(r.id, 'date', e.target.value)} className="w-full text-xs outline-none" /></td>
                                        <td className="p-1 border border-slate-200 text-center text-xs font-bold">
                                            <select value={r.time} onChange={e => update(r.id, 'time', e.target.value)} className="bg-transparent outline-none">
                                                <option>朝</option><option>昼</option><option>晩</option>
                                            </select>
                                        </td>
                                        {conditionItems.map(h => (
                                            <td key={h} className="border border-slate-200 text-center">
                                                <select value={r.conditions[h] || ''} onChange={e => update(r.id, 'conditions', e.target.value, h)} className="w-full outline-none bg-transparent text-center font-bold">
                                                    <option value=""></option>{[1,2,3,4,5].map(v => <option key={v} value={v}>{v}</option>)}
                                                </select>
                                            </td>
                                        ))}
                                        {goodSigns.map(h => <td key={h} onClick={() => toggle(r.id, 'goodSigns', h)} className={`border border-slate-200 text-center cursor-pointer ${r.goodSigns[h] ? 'bg-blue-100' : ''}`}>{r.goodSigns[h] ? '○' : ''}</td>)}
                                        {warningSigns.map(h => <td key={h} onClick={() => toggle(r.id, 'warningSigns', h)} className={`border border-slate-200 text-center cursor-pointer ${r.warningSigns[h] ? 'bg-amber-100' : ''}`}>{r.warningSigns[h] ? '○' : ''}</td>)}
                                        {badSigns.map(h => <td key={h} onClick={() => toggle(r.id, 'badSigns', h)} className={`border border-slate-200 text-center cursor-pointer ${r.badSigns[h] ? 'bg-red-100' : ''}`}>{r.badSigns[h] ? '○' : ''}</td>)}
                                        {recoveryItems.map(h => <td key={h} onClick={() => toggle(r.id, 'recovery', h)} className={`border border-slate-200 text-center cursor-pointer ${r.recovery[h] ? 'bg-green-100' : ''}`}>{r.recovery[h] ? '○' : ''}</td>)}
                                        <td onClick={() => update(r.id, 'medication', !r.medication)} className={`border border-slate-200 text-center cursor-pointer ${r.medication ? 'bg-purple-100' : ''}`}>{r.medication ? '○' : ''}</td>
                                        <td className="p-1 border border-slate-200"><input type="text" value={r.note} onChange={e => update(r.id, 'note', e.target.value)} className="w-full text-xs outline-none" /></td>
                                    </tr>
                                ))}
                            </tbody>
                        </table>
                    </div>
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
</html>
